\begin{align*}

f(x) =& x^2+3x+2 \\       % No space or automatic space for context-sensitive spacing
f(x) =& x^2\, +3x\, +2 \\ % Tiny space
f(x) =& x^2\: +3x\: +2 \\
f(x) =& x^2\; +3x\; +2 \\
f(x) =& x^2\ +3x\ +2 \\   % em space
f(x) =& x^2\quad +3x\quad +2 \\ % big space
f(x) =& x^2\qquad +3x\qquad +2  % gigantic space
\end{align*}
$$
