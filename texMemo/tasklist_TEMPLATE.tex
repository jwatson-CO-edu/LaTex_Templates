% URL, Format a counter with leading zeros: http://tex.stackexchange.com/a/30940
\usepackage{fmtcount}% http://ctan.org/pkg/fmtcount
\usepackage{intcalc}

\newcounter{lrgNum}
\setcounter{lrgNum}{100}
\newcounter{medNum}
\setcounter{medNum}{10}
\newcounter{smlNum}
\setcounter{smlNum}{1}

\newcommand{\dispTask}[1]{ % display the counter value as a decimal with leading zeros padded to 4 digits
  \padzeroes[4]{\decimal{#1}} 
}

\newcommand{\lrgIncr}[1]{ % Increment to the next main task
	\setcounter{#1}{ \intcalcMul{ \intcalcDiv{\value{#1}}{\value{lrgNum}} }{ \value{lrgNum} } }
	\addtocounter{#1}{\value{lrgNum}}
	\dispTask{#1}
}

\newcommand{\medIncr}[1]{ % Increment to the next 2nd-level task
	\setcounter{#1}{ \intcalcMul{ \intcalcDiv{\value{#1}}{\value{medNum}} }{ \value{medNum} } }
	\addtocounter{#1}{\value{medNum}}
	\dispTask{#1}
}

\newcommand{\smlIncr}[1]{ % Increment to the next 3rd-level task
	\addtocounter{#1}{1}
	\dispTask{#1}
}

\newcounter{taskCtr}
\setcounter{taskCtr}{0}

%URL, enumerate with custom levels: http://tex.stackexchange.com/a/201936
\usepackage{enumitem}
% Define a task list that gives each item its own 4-digit code
\newlist{taskList}{itemize}{3}% allow 3 levels of nesting in an enumerate-like environment
\setlist[taskList,1]{label=\lrgIncr{taskCtr}}% labels for top level
\setlist[taskList,2]{label=\medIncr{taskCtr}}% labels for top level
\setlist[taskList,3]{label=\smlIncr{taskCtr}}% labels for top level
